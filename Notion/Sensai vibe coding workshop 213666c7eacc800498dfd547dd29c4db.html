      import requests
import xml.etree.ElementTree as ET
import csv
from urllib.parse import urljoin, urlparse
import time
from typing import List, Set

class SitemapExtractor:
    def __init__(self, timeout: int = 10):
        """
        Inicjalizuje ekstraktor sitemap
        
        Args:
            timeout: Timeout dla HTTP requestów w sekundach
        """
        self.timeout = timeout
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        })
    
    def fetch_sitemap(self, url: str) -> str:
        """
        Pobiera zawartość sitemap z podanego URL
        
        Args:
            url: URL do sitemap
            
        Returns:
            Zawartość XML jako string
        """
        try:
            response = self.session.get(url, timeout=self.timeout)
            response.raise_for_status()
            return response.text
        except requests.RequestException as e:
            print(f"Błąd podczas pobierania {url}: {e}")
            return ""
    
    def parse_sitemap_index(self, xml_content: str) -> List[str]:
        """
        Parsuje sitemap index i zwraca listę URL-ów do sub-sitemaps
        
        Args:
            xml_content: Zawartość XML sitemap index
            
        Returns:
            Lista URL-ów do sub-sitemaps
        """
        try:
            root = ET.fromstring(xml_content)
            
            # Sprawdzamy różne namespace'y
            namespaces = {
                'ns': 'http://www.sitemaps.org/schemas/sitemap/0.9'
            }
            
            sitemap_urls = []
            
            # Szukamy elementów sitemap w sitemap index
            for sitemap in root.findall('.//ns:sitemap', namespaces):
                loc = sitemap.find('ns:loc', namespaces)
                if loc is not None and loc.text:
                    sitemap_urls.append(loc.text.strip())
            
            # Fallback - szukamy bez namespace
            if not sitemap_urls:
                for sitemap in root.findall('.//sitemap'):
                    loc = sitemap.find('loc')
                    if loc is not None and loc.text:
                        sitemap_urls.append(loc.text.strip())
            
            return sitemap_urls
            
        except ET.ParseError as e:
            print(f"Błąd parsowania XML sitemap index: {e}")
            return []
    
    def parse_sitemap_urls(self, xml_content: str) -> List[dict]:
        """
        Parsuje sitemap i zwraca listę URL-ów z metadanymi
        
        Args:
            xml_content: Zawartość XML sitemap
            
        Returns:
            Lista słowników z URL-ami i metadanymi
        """
        try:
            root = ET.fromstring(xml_content)
            
            # Sprawdzamy różne namespace'y
            namespaces = {
                'ns': 'http://www.sitemaps.org/schemas/sitemap/0.9'
            }
            
            urls = []
            
            # Szukamy elementów url w sitemap
            for url in root.findall('.//ns:url', namespaces):
                url_data = {}
                
                # Lokalizacja (wymagana)
                loc = url.find('ns:loc', namespaces)
                if loc is not None and loc.text:
                    url_data['loc'] = loc.text.strip()
                
                # Ostatnia modyfikacja (opcjonalna)
                lastmod = url.find('ns:lastmod', namespaces)
                if lastmod is not None and lastmod.text:
                    url_data['lastmod'] = lastmod.text.strip()
                
                # Częstotliwość zmian (opcjonalna)
                changefreq = url.find('ns:changefreq', namespaces)
                if changefreq is not None and changefreq.text:
                    url_data['changefreq'] = changefreq.text.strip()
                
                # Priorytet (opcjonalny)
                priority = url.find('ns:priority', namespaces)
                if priority is not None and priority.text:
                    url_data['priority'] = priority.text.strip()
                
                if 'loc' in url_data:
                    urls.append(url_data)
            
            # Fallback - szukamy bez namespace
            if not urls:
                for url in root.findall('.//url'):
                    url_data = {}
                    
                    loc = url.find('loc')
                    if loc is not None and loc.text:
                        url_data['loc'] = loc.text.strip()
                    
                    lastmod = url.find('lastmod')
                    if lastmod is not None and lastmod.text:
                        url_data['lastmod'] = lastmod.text.strip()
                    
                    changefreq = url.find('changefreq')
                    if changefreq is not None and changefreq.text:
                        url_data['changefreq'] = changefreq.text.strip()
                    
                    priority = url.find('priority')
                    if priority is not None and priority.text:
                        url_data['priority'] = priority.text.strip()
                    
                    if 'loc' in url_data:
                        urls.append(url_data)
            
            return urls
            
        except ET.ParseError as e:
            print(f"Błąd parsowania XML sitemap: {e}")
            return []
    

      def is_sitemap_index(self, xml_content: str) -> bool:
        """
        Sprawdza czy XML to sitemap index czy zwykły sitemap
        
        Args:
            xml_content: Zawartość XML
            
        Returns:
            True jeśli to sitemap index, False jeśli zwykły sitemap
        """
        try:
            root = ET.fromstring(xml_content)
            
            # Sprawdzamy czy istnieją elementy sitemapindex
            namespaces = {
                'ns': 'http://www.sitemaps.org/schemas/sitemap/0.9'
            }
            
            # Sprawdzamy z namespace
            if root.find('.//ns:sitemapindex', namespaces) is not None:
                return True
            if root.find('.//ns:sitemap', namespaces) is not None:
                return True
            
            # Sprawdzamy bez namespace
            if root.find('.//sitemapindex') is not None:
                return True
            if root.tag == 'sitemapindex' or 'sitemapindex' in root.tag:
                return True
            
            return False
            
        except ET.ParseError:
            return False
    
    def extract_urls_from_domain(self, domain: str) -> List[dict]:
        """
        Wyciąga wszystkie URL-e z sitemap danej domeny
        
        Args:
            domain: Domena (np. 'https://example.com' lub 'https://example.com/')
            
        Returns:
            Lista słowników z URL-ami i metadanymi
        """
        # Normalizujemy domenę
        if not domain.startswith(('http://', 'https://')):
            domain = 'https://' + domain
        
        if not domain.endswith('/'):
            domain += '/'
        
        # Konstruujemy URL do sitemap
        sitemap_url = urljoin(domain, 'sitemap.xml')
        
        print(f"Pobieranie sitemap z: {sitemap_url}")
        
        # Pobieramy główny sitemap
        xml_content = self.fetch_sitemap(sitemap_url)
        if not xml_content:
            return []
        
        all_urls = []
        
        # Sprawdzamy czy to sitemap index
        if self.is_sitemap_index(xml_content):
            print("Wykryto sitemap index, pobieranie sub-sitemaps...")
            
            # Pobieramy URL-e do sub-sitemaps
            sub_sitemap_urls = self.parse_sitemap_index(xml_content)
            print(f"Znaleziono {len(sub_sitemap_urls)} sub-sitemaps")
            
            # Pobieramy każdy sub-sitemap
            for i, sub_url in enumerate(sub_sitemap_urls, 1):
                print(f"Pobieranie sub-sitemap {i}/{len(sub_sitemap_urls)}: {sub_url}")
                
                sub_xml_content = self.fetch_sitemap(sub_url)
                if sub_xml_content:
                    urls = self.parse_sitemap_urls(sub_xml_content)
                    all_urls.extend(urls)
                    print(f"  Znaleziono {len(urls)} URL-ów")
                
                # Mała przerwa między requestami
                time.sleep(0.1)
        
        else:
            print("Wykryto zwykły sitemap, parsowanie URL-ów...")
            # To zwykły sitemap, parsujemy URL-e bezpośrednio
            all_urls = self.parse_sitemap_urls(xml_content)
        
        print(f"Łącznie znaleziono {len(all_urls)} URL-ów")
        return all_urls
    
    def extract_urls_from_sitemap_url(self, sitemap_url: str) -> List[dict]:
        """
        Wyciąga URL-e z konkretnego URL sitemap
        
        Args:
            sitemap_url: Bezpośredni URL do sitemap
            
        Returns:
            Lista słowników z URL-ami i metadanymi
        """
        print(f"Pobieranie sitemap z: {sitemap_url}")
        
        xml_content = self.fetch_sitemap(sitemap_url)
        if not xml_content:
            return []
        
        all_urls = []
        
        if self.is_sitemap_index(xml_content):
            print("Wykryto sitemap index, pobieranie sub-sitemaps...")
            
            sub_sitemap_urls = self.parse_sitemap_index(xml_content)
            print(f"Znaleziono {len(sub_sitemap_urls)} sub-sitemaps")
            
            for i, sub_url in enumerate(sub_sitemap_urls, 1):
                print(f"Pobieranie sub-sitemap {i}/{len(sub_sitemap_urls)}: {sub_url}")
                
                sub_xml_content = self.fetch_sitemap(sub_url)
                if sub_xml_content:
                    urls = self.parse_sitemap_urls(sub_xml_content)
                    all_urls.extend(urls)
                    print(f"  Znaleziono {len(urls)} URL-ów")
                
                time.sleep(0.1)
        else:
            print("Wykryto zwykły sitemap, parsowanie URL-ów...")
            all_urls = self.parse_sitemap_urls(xml_content)
        
        print(f"Łącznie znaleziono {len(all_urls)} URL-ów")
        return all_urls
    
    def save_to_csv(self, urls: List[dict], filename: str = 'sitemap_urls.csv'):
        """
        Zapisuje URL-e do pliku CSV
        
        Args:
            urls: Lista słowników z URL-ami
            filename: Nazwa pliku wyjściowego
        """
        if not urls:
            print("Brak URL-ów do zapisania")
            return
        
        # Zbieramy wszystkie możliwe klucze
        fieldnames = set()
        for url in urls:
            fieldnames.update(url.keys())
        
        fieldnames = sorted(list(fieldnames))
        
        with open(filename, 'w', newline='', encoding='utf-8') as csvfile:
            writer = csv.DictWriter(csvfile, fieldnames=fieldnames)
            writer.writeheader()
            writer.writerows(urls)
        
        print(f"Zapisano {len(urls)} URL-ów do pliku: {filename}")
    
    def save_to_txt(self, urls: List[dict], filename: str = 'sitemap_urls.txt'):
        """
        Zapisuje tylko URL-e do pliku tekstowego (jeden URL na linię)
        
        Args:
            urls: Lista słowników z URL-ami
            filename: Nazwa pliku wyjściowego
        """
        if not urls:
            print("Brak URL-ów do zapisania")
            return
        
        with open(filename, 'w', encoding='utf-8') as f:
            for url_data in urls:
                if 'loc' in url_data:
                    f.write(url_data['loc'] + '\n')
        
        print(f"Zapisano {len(urls)} URL-ów do pliku: {filename}")

// Funkcja główna (bo Node.js wymaga async/await dla Excel)
async function analizujKeywords() {
  const results = [];

  // 1. Wczytanie CSV
  await new Promise((resolve, reject) => {
    fs.createReadStream('keywords.csv')
      .pipe(csv())
      .on('data', (data) => {
        // Konwersja pozycji na liczbę (csv-parser zwraca stringi)
        data.pozycja = parseInt(data.pozycja);
        results.push(data);
      })
      .on('end', resolve)
      .on('error', reject);
  });

  // 2. Filtrowanie top 10
  const top10 = results.filter(row => row.pozycja <= 10);

  // 3. Zapis do Excel
  const workbook = new ExcelJS.Workbook();
  const worksheet = workbook.addWorksheet('Top 10');

  // Dodanie nagłówków (zakładamy że CSV ma kolumny: keyword, pozycja, klikniecia)
  worksheet.columns = [
    { header: 'Keyword', key: 'keyword', width: 30 },
    { header: 'Pozycja', key: 'pozycja', width: 10 },
    { header: 'Kliknięcia', key: 'klikniecia', width: 15 }
  ];

  // Dodanie danych
  top10.forEach(row => {
    worksheet.addRow(row);
  });

  // Zapisanie pliku
  await workbook.xlsx.writeFile('raport.xlsx');

  console.log('Gotowe! Zapisano do raport.xlsx');
}

// Uruchomienie z obsługą błędów

  console.error('Błąd:', error);

✅ Security Rules (wklej to do .cursora)

authentication:
- ZAWSZE weryfikuj tokeny przed użyciem - nigdy nie ufaj inputowi użytkownika
- NIGDY nie wysyłaj tokenów w URL/query params - używaj Authorization headers
- NIGDY nie loguj pełnych tokenów - maksymalnie pierwsze/ostatnie 4 znaki
- Używaj krótkich czasów wygaśnięcia tokenów (30 min dla access, 7 dni dla refresh)
- Implementuj token rotation i refresh token flow
- Hashuj hasła używając bcrypt/argon2 z odpowiednim salt

database_security:
- ZAWSZE używaj Row Level Security (RLS) w Supabase/PostgreSQL
- NIGDY nie używaj service/admin keys w kodzie klienckim
- ZAWSZE używaj parametryzowanych zapytań - nigdy string concatenation
- Szyfruj wrażliwe dane przed zapisem do bazy (PII, tokeny, klucze API)
- Regularnie rotuj klucze dostępowe do bazy danych
- Ogranicz uprawnienia użytkowników bazy do minimum (principle of least privilege)

input_validation:
- ZAWSZE waliduj i sanityzuj wszystkie inputy użytkownika
- Sprawdzaj typy danych, długość, format, dozwolone znaki
- Odrzucaj podejrzane wzorce (Path Traversal: ../, SQL: '; DROP TABLE)
- Używaj whitelist zamiast blacklist dla dozwolonych wartości
- Escapuj specjalne znaki przed wyświetleniem (XSS prevention)
- Waliduj zarówno po stronie klienta jak i serwera

api_security:
- Implementuj rate limiting na poziomie użytkownika i IP
- Używaj CORS z konkretnymi originami - nigdy wildcard (*)
- ZAWSZE używaj HTTPS - nigdy plain HTTP w produkcji
- Dodaj security headers (CSP, X-Frame-Options, X-Content-Type-Options)
- Implementuj request signing dla krytycznych operacji
- Monitoruj i loguj podejrzane aktywności

error_handling:

- Używaj generycznych komunikatów błędów dla użytkowników
- Loguj szczegóły błędów tylko po stronie serwera
- Nie ujawniaj struktury bazy danych w błędach
- Nie pokazuj wersji frameworków/bibliotek w błędach

- Używaj zmiennych środowiskowych dla konfiguracji

ON profiles FOR UPDATE
USING ((SELECT auth.uid()) = id);

ON profiles FOR DELETE
USING ((SELECT auth.uid()) = id);

TO authenticated

TO authenticated

// middleware.ts
import { createServerClient } from '@supabase/ssr'
import { NextResponse, type NextRequest } from 'next/server'

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({
    request,
  })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll()
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value, options }) => {
            request.cookies.set(name, value)
            supabaseResponse.cookies.set(name, value, options)
          })
        },
      },
    }
  )

  // IMPORTANT: Avoid writing any logic between createServerClient and
  // supabase.auth.getUser(). A simple mistake could make it very hard to debug
  // issues with users being randomly logged out.

  const {
    data: { user },
  } = await supabase.auth.getUser()

  if (
    !user &&
    !request.nextUrl.pathname.startsWith('/login') &&
    !request.nextUrl.pathname.startsWith('/auth')
  ) {
    // no user, potentially respond by redirecting the user to the login page
    const url = request.nextUrl.clone()
    url.pathname = '/login'
    return NextResponse.redirect(url)
  }

  // IMPORTANT: You *must* return the supabaseResponse object as it is. If you're
  // creating a new response object with NextResponse.next() make sure to:
  // 1. Pass the request in it, like so:
  //    const myNewResponse = NextResponse.next({ request })
  // 2. Copy over the cookies, like so:
  //    myNewResponse.cookies.setAll(supabaseResponse.cookies.getAll())
  // 3. Change the myNewResponse object instead of the supabaseResponse object

  return supabaseResponse
}

CREATE OR REPLACE FUNCTION match_documents(
  query_embedding VECTOR(1536),
  match_threshold FLOAT,
  match_count INT
)
RETURNS SETOF documents
LANGUAGE sql STABLE
AS $$
  SELECT *
  FROM documents
  WHERE documents.embedding <=> query_embedding < 1 - match_threshold
  ORDER BY documents.embedding <=> query_embedding
  LIMIT match_count;
$$;

Express Middleware Setup

import fs from 'node:fs'
import path from 'node:path'
import { fileURLToPath } from 'node:url'
import express from 'express'
import { createServer as createViteServer } from 'vite'

const __dirname = path.dirname(fileURLToPath(import.meta.url))

async function createServer() {
  const app = express()

  const vite = await createViteServer({
    server: { middlewareMode: true },
    appType: 'custom'
  })

  app.use(vite.middlewares)

  app.use('*all', async (req, res) => {
    try {
      // 1. Read index.html
      let template = fs.readFileSync(
        path.resolve(__dirname, 'index.html'),
        'utf-8',
      )

      // 2. Apply Vite HTML transforms
      template = await vite.transformIndexHtml(req.originalUrl, template)

      // 3. Load the server entry
      const { render } = await vite.ssrLoadModule('/src/entry-server.js')

      // 4. Render the app HTML
      const appHtml = await render(req.originalUrl)

      // 5. Inject the app-rendered HTML into the template
      const html = template.replace(`<!--ssr-outlet-->`, appHtml)

      // 6. Send the rendered HTML back
      res.status(200).set({ 'Content-Type': 'text/html' }).end(html)
    } catch (e) {
      vite.ssrFixStacktrace(e)
      next(e)
    }
  })

  app.listen(5173)
}

authentication:
- ZAWSZE weryfikuj tokeny przed użyciem - nigdy nie ufaj inputowi użytkownika
- NIGDY nie wysyłaj tokenów w URL/query params - używaj Authorization headers
- NIGDY nie loguj pełnych tokenów - maksymalnie pierwsze/ostatnie 4 znaki
- Używaj krótkich czasów wygaśnięcia tokenów (30 min dla access, 7 dni dla refresh)
- Implementuj token rotation i refresh token flow
- Hashuj hasła używając bcrypt/argon2 z odpowiednim salt

database_security:
- ZAWSZE używaj Row Level Security (RLS) w Supabase/PostgreSQL
- NIGDY nie używaj service/admin keys w kodzie klienckim
- ZAWSZE używaj parametryzowanych zapytań - nigdy string concatenation
- Szyfruj wrażliwe dane przed zapisem do bazy (PII, tokeny, klucze API)
- Regularnie rotuj klucze dostępowe do bazy danych
- Ogranicz uprawnienia użytkowników bazy do minimum (principle of least privilege)

input_validation:
- ZAWSZE waliduj i sanityzuj wszystkie inputy użytkownika
- Sprawdzaj typy danych, długość, format, dozwolone znaki
- Odrzucaj podejrzane wzorce (Path Traversal: ../, SQL: '; DROP TABLE)
- Używaj whitelist zamiast blacklist dla dozwolonych wartości
- Escapuj specjalne znaki przed wyświetleniem (XSS prevention)
- Waliduj zarówno po stronie klienta jak i serwera

api_security:
- Implementuj rate limiting na poziomie użytkownika i IP
- Używaj CORS z konkretnymi originami - nigdy wildcard (*)
- ZAWSZE używaj HTTPS - nigdy plain HTTP w produkcji
- Dodaj security headers (CSP, X-Frame-Options, X-Content-Type-Options)
- Implementuj request signing dla krytycznych operacji
- Monitoruj i loguj podejrzane aktywności

error_handling:
- NIGDY nie zwracaj stack traces użytkownikom
- Używaj generycznych komunikatów błędów dla użytkowników
- Loguj szczegóły błędów tylko po stronie serwera
- Nie ujawniaj struktury bazy danych w błędach
- Nie pokazuj wersji frameworków/bibliotek w błędach

sensitive_data:
- NIGDY nie commituj sekretów do repozytorium
- Używaj zmiennych środowiskowych dla konfiguracji

Zasada:
Jesteś starszym inżynierem z dużym doświadczeniem w budowaniu produkcyjnych agentów AI, automatyzacji i systemów workflow. Każde zadanie, które wykonujesz, musi bez wyjątku przebiegać według następującej procedury:

Najpierw Doprecyzuj Zakres
• Zanim napiszesz jakikolwiek kod, wypracuj dokładny plan działania.
• Potwierdź swoje rozumienie celu zadania.
• Nakreśl jasny plan pokazujący, które funkcje, moduły lub komponenty zostaną zmodyfikowane i dlaczego.
• Nie rozpoczynaj implementacji, dopóki nie wykonasz tego kroku i go nie uzasadnisz.

Zlokalizuj Dokładne Miejsce Wstawienia Kodu
• Wskaż precyzyjnie plik(i) i linie, w których umieścisz zmiany.
• Nigdy nie dokonuj rozległych edycji w niespowinowaconych plikach.
• Jeśli potrzebne są modyfikacje w wielu plikach, uzasadnij każdą z nich.
• Nie wprowadzaj nowych warstw abstrakcji ani refaktoryzacji, chyba że zadanie tego wyraźnie wymaga.

Zmiany Minimalne i Ograniczone
• Pisz tylko kod bezpośrednio niezbędny do realizacji zadania.
• Unikaj dodawania logowania, komentarzy, testów, TODO, sprzątania czy obsługi błędów, chyba że jest to konieczne.
• Żadne spekulatywne poprawki czy „przy okazji” zmiany.
• Cała logika powinna być odizolowana, by nie zaburzyć istniejących przepływów.

Podwójna Weryfikacja
• Sprawdź poprawność, zgodność z zakresem i możliwe skutki uboczne.
• Upewnij się, że Twój kod pasuje do wzorców w bazie kodu i nie wprowadza regresji.
• Zweryfikuj, czy jakiekolwiek downstreamowe komponenty nie zostaną dotknięte.

Dostarcz Klarownie
• Podsumuj, co zostało zmienione i dlaczego.
• Wypisz każdy zmodyfikowany plik i opisz w nim dokonane zmiany.
• Jeśli są jakiekolwiek założenia lub ryzyka, zaznacz je do przeglądu.

Przypomnienie: Nie jesteś współpilotem, asystentem ani partnerem do burzy mózgów. Jesteś starszym inżynierem odpowiedzialnym za decyzje wysokiego priorytetu w środowisku produkcyjnym. Nie improwizuj. Nie nadinżynieruj. Nie odbiegaj od procedury.

## Kluczowe zasady:

1. **Webhook Setup**:
   - Użyj `responseMode: "responseNode"` w parametrach Webhook node
   - NIE używaj domyślnego "When last node finishes"

2. **Struktura workflow**:
Webhook → Przygotowanie danych → Zapis do DB → Response Node → Reszta processingu

3. **Response Node**:
- Umieść "Respond to Webhook" node PO operacji która generuje dane do zwrócenia
- Jeśli zwracasz dane z DB (np. Supabase z Prefer: return=representation), 
  możesz zostawić Response node bez konfiguracji - automatycznie przekaże otrzymane dane
- Response node MUSI być w głównym flow, nie w odgałęzieniu

4. **Ważne szczegóły**:
- Dane płyną sekwencyjnie przez node'y
- Response node wykonuje się RAZ i natychmiast zwraca odpowiedź
- Wszystkie node'y PO Response node wykonują się asynchronicznie w tle
- Upewnij się, że dane potrzebne do odpowiedzi są dostępne PRZED Response node

- NIE łącz node'ów równolegle gdy potrzebujesz danych z jednego w drugim
- NIE umieszczaj Response node zbyt wcześnie - musi mieć dostęp do wszystkich danych
- NIE próbuj konfigurować skomplikowanych responseBody jeśli dane już są w pipeline

import { createContext, useContext, useEffect, useState } from "react"

type Theme = "dark" | "light" | "system"

type ThemeProviderProps = {
  children: React.ReactNode
  defaultTheme?: Theme
  storageKey?: string
}

type ThemeProviderState = {
  theme: Theme
  setTheme: (theme: Theme) => void
}

const initialState: ThemeProviderState = {
  theme: "system",
  setTheme: () => null,
}

const ThemeProviderContext = createContext<ThemeProviderState>(initialState)

export function ThemeProvider({
  children,
  defaultTheme = "system",
  storageKey = "vite-ui-theme",
  ...props
}: ThemeProviderProps) {
  const [theme, setTheme] = useState<Theme>(
    () => (localStorage.getItem(storageKey) as Theme) || defaultTheme
  )

  useEffect(() => {
    const root = window.document.documentElement

    root.classList.remove("light", "dark")

    if (theme === "system") {
      const systemTheme = window.matchMedia("(prefers-color-scheme: dark)")
        .matches
        ? "dark"
        : "light"

      root.classList.add(systemTheme)
      return
    }

    root.classList.add(theme)
  }, [theme])

  const value = {
    theme,
    setTheme: (theme: Theme) => {
      localStorage.setItem(storageKey, theme)
      setTheme(theme)
    },
  }

  return (
    <ThemeProviderContext.Provider {...props} value={value}>
      {children}
    </ThemeProviderContext.Provider>
  )
}

export const useTheme = () => {
  const context = useContext(ThemeProviderContext)

  if (context === undefined)
    throw new Error("useTheme must be used within a ThemeProvider")

import {
  ColumnDef,

  getCoreRowModel,
  useReactTable,
} from "@tanstack/react-table"

import {
  Table,
  TableBody,
  TableCell,
  TableHead,
  TableHeader,
  TableRow,
} from "@/components/ui/table"

interface DataTableProps<TData, TValue> {
  columns: ColumnDef<TData, TValue>[]
  data: TData[]
}

export function DataTable<TData, TValue>({
  columns,
  data,
}: DataTableProps<TData, TValue>) {
  const table = useReactTable({
    data,
    columns,
    getCoreRowModel: getCoreRowModel(),
  })

  return (
    <div className="rounded-md border">
      <Table>
        <TableHeader>
          {table.getHeaderGroups().map((headerGroup) => (
            <TableRow key={headerGroup.id}>
              {headerGroup.headers.map((header) => {
                return (
                  <TableHead key={header.id}>
                    {header.isPlaceholder
                      ? null
                      : flexRender(
                          header.column.columnDef.header,
                          header.getContext()
                        )}
                  </TableHead>
                )
              })}
            </TableRow>
          ))}
        </TableHeader>
        <TableBody>
          {table.getRowModel().rows?.length ? (
            table.getRowModel().rows.map((row) => (
              <TableRow
                key={row.id}
                data-state={row.getIsSelected() && "selected"}
              >
                {row.getVisibleCells().map((cell) => (
                  <TableCell key={cell.id}>
                    {flexRender(cell.column.columnDef.cell, cell.getContext())}
                  </TableCell>
                ))}
              </TableRow>
            ))
          ) : (
            <TableRow>
              <TableCell colSpan={columns.length} className="h-24 text-center">
                No results.
              </TableCell>
            </TableRow>
          )}
        </TableBody>
      </Table>
    </div>
  )
}

📞 API

API (Application Programming Interface) to interfejs programowania aplikacji — zestaw reguł i narzędzi, które pozwalają różnym programom komunikować się ze sobą.

Prosto mówiąc:

API to "kelner" między Twoją aplikacją a inną usługą. Ty składasz zamówienie (np. chcesz dane pogodowe), kelner (API) przekazuje je do kuchni (serwer pogodowy), a potem przynosi Ci odpowiedź (np. 22°C i słonecznie).

Przykład z życia:

Masz aplikację pogodową na telefonie. Ona:

  • nie zna pogody sama z siebie, tylko
  • pyta API pogodowe, np. OpenWeather,
  • API odpowiada danymi w formacie JSON (np. { "temp": 22, "desc": "sunny" }),
  • aplikacja wyświetla to Tobie na ekranie.

Rodzaje API:

  • REST API – najpopularniejsze dziś (działa przez HTTP, jak strony internetowe),
  • GraphQL API – bardziej elastyczne, możesz precyzyjnie określić, jakie dane chcesz,
  • WebSocket API – używane do danych "na żywo", np. czat albo giełda.

Gdzie spotykasz API?

  • Logowanie przez Google/Facebook – Twoja aplikacja używa API Google, by dostać dane użytkownika.
  • Mapy w aplikacjach – korzystają z API Google Maps.
  • ChatGPT – też ma swoje API, które inni mogą używać do generowania tekstów.

React

React to biblioteka JavaScript (stworzona i utrzymywana przez Facebooka/Meta) służąca do budowania interfejsów użytkownika w aplikacjach webowych.

Kluczowe cechy Reacta

  1. Deklaratywność
    • Opisujesz, jak UI powinno wyglądać w danym stanie, a React sam dba o aktualizację DOM-u, gdy stan się zmienia.
  1. Komponentowa architektura
    • Cały interfejs dzielisz na małe, wielokrotnego użytku komponenty (funkcje lub klasy), np. <Button />, <Header />, <UserCard />.
    • Komponenty mogą przyjmować „props” (parametry) i zarządzać własnym stanem (state).
  1. JSX
    • Rozszerzenie składni JavaScriptu pozwalające pisać strukturę UI w stylu XML/HTML bezpośrednio w kodzie JS/TS:
      function Greeting({ name }) {
  return <h1>Cześć, {name}!</h1>;
}
  1. Virtual DOM
    • React utrzymuje w pamięci uproszczony model drzewa DOM (tzw. Virtual DOM).
    • Przy zmianie stanu porównuje nowy i poprzedni Virtual DOM i minimalnie aktualizuje realny DOM, co jest wydajne.
  1. Ekosystem
    • Router (React Router) do nawigacji między stronami.
    • Zarządzanie stanem globalnym (Redux, MobX, Context API).
    • Hooki (od wersji 16.8) – funkcje typu useState, useEffect do zarządzania stanem i efektami ubocznymi w komponentach funkcyjnych.

Dlaczego warto używać Reacta?

  • Modularność – łatwo rozdzielić projekt na małe części.
  • Wydajność – dzięki Virtual DOM minimalizuje niepotrzebne operacje na prawdziwym DOM.
  • Ogromna społeczność – mnóstwo gotowych bibliotek, narzędzi i materiałów edukacyjnych.
  • Uniwersalność – poza webem (React DOM) są React Native (aplikacje mobilne), React VR/Three.js itp.

Podsumowanie: React to potężne, wydajne i elastyczne narzędzie do budowy nowoczesnych aplikacji interaktywnych, oparte na komponencie, JSX i wirtualnym modelu DOM.

Supabase

Supabase to otwarto-źródłowa platforma „Backend as a Service” (BaaS), często określana jako alternatywa dla Firebase, oparta na PostgreSQL. Oferuje zestaw gotowych serwisów, dzięki którym możesz szybko zbudować backend swojej aplikacji bez konieczności konfigurowania serwera od zera:

  1. Baza danych (PostgreSQL)
    • W pełni zarządzany klaster Postgresa ze wsparciem dla RLS (Row Level Security), funkcji, triggerów, rozszerzeń itp.
    • Automatycznie generowane API REST i WebSockety (PostgREST + Realtime), dzięki czemu możesz operować na tabelach przez HTTP lub subskrybować zmiany w czasie rzeczywistym.
  1. Uwierzytelnianie (Auth)
    • Obsługa logowania przez e-mail/hasło, magic links, OAuth (Google, GitHub, Twitter itd.).
    • Zarządzanie sesjami, tokenami JWT i rolami użytkowników.
  1. Storage
    • Przechowywanie plików (obrazy, dokumenty) z mechanizmem kontroli dostępu i generowaniem bezpiecznych URL-i.
  1. Realtime
    • Subskrypcje na zmiany w tabelach (INSERT/UPDATE/DELETE) przez WebSockety, przydatne do chatów, dashboardów na żywo itp.
  1. Edge Functions (Serverless)
    • Możliwość pisania i wdrażania funkcji serverless w JavaScript/TypeScript, działających blisko użytkownika. Edge Functions dają możliwość:
      • centralizacji i łatwiejszego utrzymania logiki,
      • lepszej wydajności dzięki edge-computing,
      • elastycznego łączenia kilku źródeł bez ujawniania dodatkowych kluczy (przydatne kiedy korzystamy z zewnętrznego API, np Senuto, Ahrefs etc)
  1. Dashboard & CLI
    • Przyjazny panel webowy do zarządzania bazą, tabelami, uprawnieniami i ustawieniami projektu.
    • supabase CLI do lokalnego developmentu, migracji bazy i deploya.
  1. Self-hosted lub chmura
    • Możesz korzystać z hostingu Supabase (w modelu SaaS) lub postawić własny klaster dzięki open-source’owym komponentom.

Dlaczego warto?

  • Masz pełnię mocy PostgreSQL + gotowe API.
  • Szybki start dzięki gotowym modułom Auth, Storage i Realtime.
  • Możliwość rozszerzania przez własne funkcje i rozszerzenia Postgresa.
  • Duża społeczność i szybki rozwój projektu.

Supabase pozwala skupić się na tworzeniu frontendu i logiki biznesowej, zamiast konfigurować i utrzymywać infrastrukturę backendową.

Co to jest Docker?

Wyobraź sobie, że masz idealnie skonfigurowany komputer do analizy SEO:

  • Zainstalowany Python
  • Wszystkie biblioteki (pandas, requests, etc.)
  • Twój skrypt SEO
  • Ustawienia środowiskowe

Docker to jak "zapisanie" całego tego komputera w pudełku, które możesz uruchomić WSZĘDZIE - na swoim laptopie, serwerze, u kolegi.

Analogia dla SEOwca:

🏠 Tradycyjne podejście:
- Instalujesz Python na serwerze
- Instalujesz biblioteki
- Kopiujesz skrypt
- Modlisz się żeby działało...
- Na innym serwerze wszystko od nowa 😱

📦 Docker:
- Pakujesz wszystko w "kontener"
- Wysyłasz kontener na serwer
- Uruchamiasz jedną komendą
- Działa IDENTYCZNIE jak u Ciebie! 🎉

Jak to wygląda w praktyce?

1. Dockerfile = Przepis na Twój kontener

# Zacznij od Pythona
FROM python:3.9

# Skopiuj listę bibliotek
COPY requirements.txt .

# Zainstaluj biblioteki
RUN pip install -r requirements.txt

# Skopiuj Twój skrypt
COPY seo_analyzer.py .

# Uruchom skrypt
CMD ["python", "seo_analyzer.py"]

To jak przepis kucharski - krok po kroku co ma być w kontenerze.

2. Docker Image = Gotowe pudełko

# "Upiecz" pudełko według przepisu
docker build -t moj-seo-tool .

Teraz masz gotowy "obraz" swojego narzędzia.

3. Docker Container = Działające narzędzie

# Uruchom pudełko
docker run moj-seo-tool

Twój skrypt działa w izolowanym środowisku!

  • Strategia Q&A (pytań i odpowiedzi)

    Zamiast natychmiast dawać odpowiedź, AI najpierw zadaje pytania doprecyzowujące. To trochę jak programowanie w parach z kimś, kto nie wyciąga pochopnych wniosków.

    Jak to działa:

    • Prezentujesz swój problem lub żądanie.
    • Zamiast natychmiast rozwiązywać problem, AI zadaje pytania, aby wyjaśnić wymagania.
    • Podajesz dodatkowe szczegóły na podstawie tych pytań.
    • AI wykorzystuje te szczegóły, aby dostarczyć bardziej dopasowane rozwiązanie.

    Korzyści:

    • Unika błędnych założeń prowadzących do niewłaściwych rozwiązań.
    • Odkrywa wymagania, o których mogłeś zapomnieć wspomnieć.
    • Generuje dokładniejsze i bardziej użyteczne odpowiedzi.
    • Wprowadza współpracę w rozwiązywaniu problemów.

    Przykład użycia:
    Zamiast pisać:

    „Pomóż mi stworzyć system uwierzytelniania użytkowników”

    Użyj strategii Q&A:

    „Potrzebuję stworzyć system uwierzytelniania użytkowników dla mojej aplikacji webowej. Zanim podasz rozwiązanie, zadaj mi pytania dotyczące moich specyficznych wymagań i ograniczeń, aby doradzić najlepszą implementację.”

    Kiedy podejmujemy decyzję architektoniczną której nie jesteśmy pewni

    Jak to działa:

    • Przedstawiasz punkt decyzyjny lub wybór między kilkoma opcjami.

    • Zapobiega jednostronnym rekomendacjom.
    • Wymusza uwzględnienie potencjalnych wad.

    Przykład użycia:

    • Pełna kontrola nad tempem i kierunkiem pracy.
    • Możliwość korekt na każdym etapie.
    • Zapobiega narastaniu błędów.

Oto przewodnik krok po kroku:

Krok 1: Przygotowanie w Google Cloud Platform (GCP)

  1. Utwórz lub wybierz projekt GCP:
    • Przejdź do Google Cloud Console i utwórz nowy projekt.
  1. Włącz Google Search Console API:
    • W konsoli GCP, w menu nawigacyjnym po lewej stronie, przejdź do "Interfejsy API i usługi" > "Biblioteka".
    • W polu wyszukiwania wpisz "Google Search Console API".
    • Kliknij na znalezione API i następnie kliknij przycisk "WŁĄCZ".

Krok 2: Tworzenie danych logowania (Konto Usługi)

Dla skryptów i aplikacji działających po stronie serwera (jak Twój skrypt Pythona) najlepszym sposobem autoryzacji jest użycie Konta Usługi.

  1. Przejdź do tworzenia danych logowania:
    • W konsoli GCP, w menu nawigacyjnym, przejdź do "Interfejsy API i usługi" > "Dane logowania".
  1. Utwórz Konto Usługi:
    • Kliknij przycisk "+ UTWÓRZ DANE LOGOWANIA" u góry strony i z listy wybierz "Konto usługi".
    • Nazwa konta usługi: Wpisz nazwę dla swojego konta usługi, np. czytnik-gsc-planetatortow.
    • Identyfikator konta usługi: Zostanie wygenerowany automatycznie na podstawie nazwy. Możesz go zostawić lub dostosować.
    • Opis: Dodaj opcjonalny opis, np. "Konto do odczytu danych GSC dla Planety Tortów przez API".
    • Kliknij "UTWÓRZ I KONTYNUUJ".
  1. Nadaj uprawnienia (w GCP - opcjonalne na tym etapie):
    • W sekcji "Przyznaj temu kontu usługi dostęp do projektu" możesz na razie pominąć nadawanie ról w GCP, klikając "KONTYNUUJ". Uprawnienia do konkretnej usługi GSC nadasz później bezpośrednio w interfejsie Search Console.

python
from flask import Flask, jsonify, request, send_file, g
from flask_cors import CORS
from functools import wraps
import os
from supabase import create_client

app = Flask(__name__)
CORS(app)

# Supabase client
supabase = create_client(
    os.getenv('SUPABASE_URL'),
    os.getenv('SUPABASE_SERVICE_KEY')# Service key dla RLS bypass w backendzie
)

def require_auth(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        auth_header = request.headers.get('Authorization', '')
        if not auth_header.startswith('Bearer '):
            return jsonify({'error': 'No token provided'}), 401

        token = auth_header.split(' ')[1]
        try:
# Weryfikacja tokenu przez Supabase
            user = supabase.auth.get_user(token)
            g.user = user
        except Exception:
            return jsonify({'error': 'Invalid token'}), 401

        return f(*args, **kwargs)
    return decorated_function

@app.route('/analyze', methods=['POST'])
@require_auth
def analyze():
    domain = request.json.get('domain')
    user_id = g.user.id

# Uruchom analizę
    results = run_seo_analysis(domain)

# Dodaj user_id do każdego wyniku
    for result in results:
        result['user_id'] = user_id

# Zapisz do Supabase
    supabase.table('opportunities').insert(results).execute()

    return jsonify({
        'status': 'success',
        'count': len(results),
        'domain': domain
    })

@app.route('/results/<domain>')
@require_auth
def get_results(domain):
    user_id = g.user.id

    response = supabase.table('opportunities')\
        .select('*')\
        .eq('user_id', user_id)\
        .eq('domain', domain)\
        .order('priority_score', desc=True)\
        .execute()

    return jsonify(response.data)

@app.route('/health')
def health():
    return jsonify({'status': 'ok'})

Create a professional SEO Opportunity Finder web application with the following specifications:

CORE FUNCTIONALITY

This tool helps SEO specialists find keyword opportunities by analyzing domains. Users log in via Google, enter a domain, wait 2-3 minutes for analysis, and receive a sorted list of SEO opportunities with Priority Scores that they can export to Excel.

AUTHENTICATION (Supabase)

  • Use Bolt.new's Supabase integration
  • Google OAuth only (no email/password)
  • After login, store user session and show email in top-right
  • Logout button next to email
  • Protected routes - redirect to login if not authenticated

CRITICAL DESIGN REQUIREMENTS - SUPABASE DARK THEME

The entire application MUST follow Supabase's dark theme design system:

Color Palette:

  • Background: #0a0a0a (pure dark)
  • Card/Panel background: #18181b with subtle border: 1px solid rgba(255,255,255,0.1)
  • Primary accent: #2ba6ff (blue for CTAs and success states)
  • Text primary: #ffffff
  • Text secondary: #a1a1aa
  • Border color: #27272a
  • Hover states: add rgba(255,255,255,0.05) overlay
  • NO GLASSMORPHISM - use solid dark backgrounds

Typography:

  • Font: Inter or system-ui
  • Headings: font-semibold text-white
  • Body text: text-gray-400
  • Small text: text-xs text-gray-500

Components Style:

  • Buttons:
    • Primary: bg-blue-500 hover:bg-blue-600 text-white px-4 py-2 rounded-md (#2ba6ff base color)
    • Secondary: bg-gray-800 hover:bg-gray-700 text-gray-300 border border-gray-700
  • Inputs: bg-gray-900 border border-gray-800 rounded-md px-3 py-2 text-white placeholder-gray-500
  • Cards: bg-gray-900 border border-gray-800 rounded-lg p-6
  • Tables: Dark theme with hover states like Supabase Table Editor
